/*
 * The following is a class for hello world but contains
 * functions for basic arithmetic operations.
 */

public class Main {
    public static void main(String[] args) {
        System.out.println("Hello World!");
        test();
    }

    public static void add(int a, int b) {
        return a + b;
    }

    public static void subtract(int a, int b) {
        return a - b;
    }

    public static void multiply(int a, int b) {
        return a * b;
    }

    public static void divide(int a, int b) {
        return a / b;
    }

    public static void mod(int a, int b) {
        return a % b;
    }

    public static void power(int a, int b) {
        int result = a;
        for (int i = 0; i < b - 1; i++) {
            result *= a;
        }
        return result;
    }

    public static void test() {
        // Test all the functions
        int a = 20;
        int b = 5;

        int resultAdd = add(a, b);
        int resultSubtract = subtract(a, b);
        int resultMultiply = multiply(a, b);
        int resultDivide = divide(a, b);
        int resultMod = mod(a, b);
        int resultPower = power(a, b);

        System.out.println("Result of add was " + resultAdd.toString());
        System.out.println("Result of subtract was " + resultSubtract.toString());
        System.out.println("Result of multiply was " + resultMultiply.toString());
        System.out.println("Result of divide was " + resultDivide.toString());
        System.out.println("Result of mod was " + resultMod.toString());
        System.out.println("Result of power was " + resultPower.toString());
    }
}
